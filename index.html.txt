    <script>
        // --- إعدادات الربط (بناءً على صورتك) ---
        const S_URL = 'https://stqpublftpzvwpbrklow.supabase.co'; 
        const S_KEY = 'sb_publishable_axv_1RBaGcMjb62L_2sX8A_JbP5wtmYy1q39N'; // تأكد من نسخ المفتاح كاملاً
        const supabaseClient = supabase.createClient(S_URL, S_KEY);

        // --- قاعدة البيانات المحلية ---
        const db = new Dexie("MansouraDB");
        db.version(1).stores({ expenses: '++id, description, amount, sync_status' });

        // --- المزامنة ---
        async function sync() {
            if (!navigator.onLine) {
                document.getElementById('sync-status').className = 'w-3 h-3 rounded-full bg-orange-500';
                return;
            }
            document.getElementById('sync-status').className = 'w-3 h-3 rounded-full bg-green-500';
            
            const pending = await db.expenses.where('sync_status').equals('pending').toArray();
            for (let item of pending) {
                const { error } = await supabaseClient.from('expenses').insert([{
                    description: item.description,
                    amount: item.amount,
                    category: item.category,
                    date: item.date
                }]);
                if (!error) await db.expenses.update(item.id, { sync_status: 'synced' });
            }
            render();
        }

        // --- إضافة مصروف ---
        document.getElementById('exp-form').onsubmit = async (e) => {
            e.preventDefault();
            const newItem = {
                description: document.getElementById('d').value,
                amount: parseFloat(document.getElementById('a').value),
                category: document.getElementById('c').value,
                date: new Date().toLocaleString('ar-EG'),
                sync_status: 'pending'
            };
            await db.expenses.add(newItem);
            document.getElementById('exp-form').reset();
            sync();
            render();
        };

        // --- عرض البيانات ---
        async function render() {
            const list = document.getElementById('list');
            const items = await db.expenses.reverse().toArray();
            let total = 0;
            list.innerHTML = '';
            items.forEach(i => {
                total += i.amount;
                const div = document.createElement('div');
                div.className = 'bg-white p-4 rounded-xl shadow-sm border flex justify-between items-center';
                div.innerHTML = `<div><p class="font-bold">${i.description}</p><p class="text-[10px] text-gray-400">${i.date} <i class="fas ${i.sync_status==='synced'?'fa-check sync-done':'fa-clock sync-pending'}"></i></p></div><p class="font-bold text-blue-600">${i.amount} ج.م</p>`;
                list.appendChild(div);
            });
            document.getElementById('total-val').innerText = total + ' ج.م';
        }

        setInterval(sync, 10000); // مزامنة كل 10 ثواني
        render();
    </script>
